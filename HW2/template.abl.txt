" LOW ORDER DIGIT
Q0  := (!Q1 & !Q0) # (Q1 & Q0) 

Q1  := ((!Q2 & Q1 & Q0) & !DOWN)) 
        # (((!Q3 & Q1 & Q0) # (Q3 & !Q1 & Q0) # (Q2 & !Q1 & !Q0)) & DOWN)

Q2  := (((!Q3 & Q2 & !Q1) # (!Q3 & !Q2 & Q1 & Q0) # (!Q3 & Q2 & Q1 & !Q0)) & !DOWN) 
        # (((Q2 & Q0) # (Q2 & Q1)) & DOWN)

Q3  := (((!Q3 & Q2 & Q1) # (Q3 & !Q2 & !Q1 & !Q0)) & !DOWN) 
        # ((Q3 & Q1) & !DOWN)

CountEn1 = ((Q3 & !Q2 & !Q1 & Q0) & DOWN) # ( (!Q3 & !Q2 & !Q1 & !Q0) & DOWN)              " enable at 9 if !DOWN, at 0 if DOWN 

" MIDDLE DIGIT

Q4  := (CountEn1 &) # (!CountEn1 &)

Q5  := (CountEn1 &) # (!CountEn1 &)

Q6  := (CountEn1 &) # (!CountEn1 &)
  
Q7  := (CountEn1 &) # (!CountEn1 &)

" HIGH ORDER DIGIT  

Q8  :=  (CountEn2 &) # (!CountEn2 &)
Q9  :=  (CountEn2 &) # (!CountEn2 &)
Q10 :=  (CountEn2 &) # (!CountEn2 &)
Q11 :=  (CountEn2 &) # (!CountEn2 &)
